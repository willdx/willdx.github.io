# -*- coding: utf-8 -*-
    '''
    File Name: ctablib.py
    Author: WillDX
    mail: xiang.dai@shuyun.com
    Created Time: 2015年10月23日 星期五 09时31分56秒
    '''
    import sys  
    import json
    import tablib  
    
    reload(sys)  
    sys.setdefaultencoding("utf-8")  
    
    class ClassTablib():
        '''python数据导出类tablib简单封装'''
    
        def __init__(self,data,headers):
            self.ob = tablib.Dataset(*data, headers=headers)
            #返回数据格式
            self.r_xlsx = self.ob.xlsx
            self.r_xls =  self.ob.xls
            self.r_json = self.ob.json
            self.r_csv =  self.ob.csv
    
        #返回数据修改
        def add_row(self,row):
            '''增加行:append([列数据])'''
            res = self.ob.append(row)
            return res
    
        def add_column(self,column):
            '''增加列:append_col([每行数据],header=列名称)'''
            res = self.ob.append_col(column)
            return res
    
        def del_row(self,start,end):
            '''删除行'''
            del self.ob[start,end]
    
        def del_column(self,c_name):
            '''删除列'''
            del self.ob[c_name]
        
        #数据写入文件
        def utf8(self, rdata, outfile):
            '''将数据写入utf-8格式文件,'''
            f = open(outfile, 'w')
            f.write(rdata)
            f.close()
    
        def gbk(self, rdata, outfile):
            '''将数据写入gbk格式文件'''
            f = open(outfile, 'w')
            f.write(rdata.decode('utf-8').encode('gb2312'))
            f.close()
    
        def binary(self, rdata, outfile):
            '''将数据写入excel文件
            因为excel表有二进制数据，所以必须要用二进制模式打开文件，即wb'''
            f = open(outfile, 'wb')
            f.write(rdata)
            f.close()
    
    if __name__=='__main__':
        #from ctablib import *

        outfile = 'json.txt'
        data = [
            ('代翔', '23'),

            ]
        obj = ClassTablib(data,headers)
        rdata = obj.r_json()
        print isinstance(rdata, str)
        #字符串-python对象-json字符串
        rdata = json.loads(rdata)
        rdata = json.dumps(rdata,indent=2,ensure_ascii=False)

    def get_productsconn(p_name):
        '''调用CMDBAPI获取产品相关的连接信息'''
        cmdb = Cmdb()
        params = {"product_name": p_name}
        res = cmdb.productsconn.get(params=params)
        return res[0]
    
    def run_comm(db_info,sql):
        '''获取命令执行的结果'''
        host = db_info['db_host']
        user = db_info['db_user']
        passwd = db_info['db_passwd']
        db_name = db_info['db_name']
        with DatabaseOperation(host, user, passwd, db_name, 3306) as mysql:
            cur = mysql.get_dict_cursor()
            res = mysql.run(cur, sql)
            return  res
    
    def format_res(res):
        '''格式化数据,将数据分解为头部和数据内容'''
        data = []
        for i in res:
            headers = i.keys()
            data.append(i.values())
        return headers,data
    
    def export_data(data,headers,outfile,f_type,f_code):
        '''根据用户输入的文件类型和编码格式导出数据'''
        obj = ClassTablib(data,headers)
        #文件类型,返回相应的数据rdata
        ty = 'r_%s' % f_type

        #编码格式,根据编码格式将rdata写入文件outfile
        if f_type in ['xls', 'xlsx']:
