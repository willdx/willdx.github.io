个推-第三方消息推送

Posted April 07, 2017

[TOC]

1.简介

个推官方文档地址

1.0 自带好用的管理后台

功能完备的管理后台, 运营人员能方便的进行各种类型的推送;

1.1 客户端多平台SDK支持

1.支持Android和iOS两大平台

1.2 多种语言版本的服务端SDK支持

提供JAVA、C#、PHP、Python等多种语言版本的服务端API SDK

如果要实现类似IM的点对点消息, 或者用户触发了什么操作, 发送一个推送消息, 需要在服务端进行集成;
个推的推送设计是模块化的, 分为消息推送模板 和 消息推送方式, 通过模板和推送的组合来满足各种场景的推送需求;

1.2.1 消息推送模板

Raw
1.点击通知打开应用模板
2.点击通知打开网页模板
3.点击通知栏弹框下载模板
4.透传消息模板

1.2.2 消息推送方式:(只说最常用的3种)

Raw
1.对单个用户推送接口
2.对指定用户列表推送接口
3.对指定应用群推接口

注意: 为了最大程度提高消息推送性能，第三方开发者需要根据业务需求合理选择消息推送形式。如果是针对每 个用户进行定制化的消息推送、或是实现类似IM的点对点消息，请采用单推消息形式（SingleMessage）；如果需要根据特定条件筛选出一批CID 后推送相同的内容，请选择批量推送形式（ListMessage）；如果希望针对省市或全量用户进行推送，请选择群推形式（AppMessage）

2. 架构图解

#个推架构图解

安卓和IOS的系统不同, 所以推送实现的逻辑不太一样:

2.1 安卓推送实现逻辑

client后台与个推云服务器保持Socket常连接，当网络不稳定时，个推SDK服务处于离线状态，待网络条件恢复后，会立即重连上线，并且重新接收在离线期间服务端推送的消息;

2.2 IOS推送实现逻辑

受 iOS系统限制，个推SDK服务在iOS平台上无法长期后台运行。当应用在前台运行时，个推SDK服务可以正常运行，可以实时接收服务端的推送消息；当应 用在后台期间，个推服务端会自动借助Apple Push Notification Service (APNs)实现消息推送，用户启动应用后，个推SDK服务可以重新接收在应用停止期间服务端推送的离线消息;

3. Python服务端集成

由于主要实现APP内部的触发式的消息推送, 所以使用 透传消息模板 + 单推 或 多推, 又由于安卓和IOS的推送方式不同, 所以在给特定用户推送透传消息时, 需要知道客户端的系统类型, 如果没有获取到系统类型, 按默认的透传去发送;

        apnpayload.sound = ""
        apnpayload.contentAvailable = 1
        apnpayload.category = "ACTIONABLE"
        apnpayload.alertMsg = alertMsg

        # 3.设定模板的APN属性
        template.setApnInfo(apnpayload)
    return template

class GtPush(object):
    """个推推送类"""

    def __init__(self, template):
        """
        :param: template: obj 模板类, 最常用TransmissionTemplateDemo

        注释: TransmissionTemplateDemo, LinkTemplateDemo, 
        NotyPopLoadTemplateDemo, NotyPopLoadTemplateDemo
        """
        print(options.GT_HOST, options.GT_APPKEY, options.GT_MASTERSECRET)
        self.push = IGeTui(options.GT_HOST, options.GT_APPKEY, options.GT_MASTERSECRET)
        self.template = template

    def pushMessageToSingle(self, cid, alias=''):
        """
        对单个用户进行推送
        :param: cid: str: 个推用户唯一标示
        """
        # 定义消息体属性
        template = self.template
        message = IGtSingleMessage()
        message.isOffline = True
        message.offlineExpireTime = 3600 * 24
        message.data = template
        message.pushNetWorkType = 0

        # 定义推送目标
        target = Target()
        target.appId = options.GT_APPID
        target.clientId = cid
        target.alias = alias

        # 推送方法
        try:
            ret = self.push.pushMessageToSingle(message, target)
        except RequestException, e:
            # 发生异常重新发送
            requstId = e.getRequestId()
            ret = self.push.pushMessageToSingle(message, target, requstId)
        return ret

    def pushMessageToList(self, cids):
        """
        对指定用户列表推送接口

        :param: cids: list: cid组成的列表,建议不超过50个
        """
        # 设定消息模板
        template = self.template
        message = IGtListMessage()
        message.data = template
        message.isOffline = True
        message.offlineExpireTime = 3600 * 24 * 7
        message.pushNetWorkType = 0

        # 组装推送消息体
        arr = []
        for cid in cids:
            target = Target()
            target.appId = options.GT_APPID
            target.clientId = cid
            arr.append(target)

        # 设定推送方式推送消息体
        contentId = self.push.getContentId(message)
        ret = self.push.pushMessageToList(contentId, arr)
        return ret

    def push_Message_ToApp(self, oslist=["ANDROID", "IOS"], plist=["上海"], taglist=[]):
        """
        指定应用全推

        :param: oslist: list: 推送平台, 默认["ANDROID", "IOS"]
        :param: plist: list: 省份标识, 默认["上海"]
        :param: taglist: list: 标签列表, 默认为空
        """
        template = self.template
        #设置是否离线，离线有效时间，推送模板，推送速度等
        message = IGtAppMessage()
        #设置消息推送速度，单位为条/秒，例如填写100，则为100条/秒。仅支持对指定应用群推接口
        message .setSpeed(100) 
        message.data = template
        #设置是否根据WIFI推送消息，1为wifi推送，0为不限制推送
        message.pushNetWorkType = 0
        message.isOffline = True
        message.offlineExpireTime = 1000 * 3600 * 12
        message.appIdList.extend([options.GT_APPID])
        message.phoneTypeList.extend(oslist)
        message.provinceList.extend(plist)
        message.tagList.extend(taglist)
        res = self.push.pushMessageToApp(message)
        print(res)
        return True

from godbox.core.gt import TransmissionTemplateDemo, LinkTemplateDemo, \
    NotyPopLoadTemplateDemo, NotyPopLoadTemplateDemo, NotificationTemplateDemo, GtPush

def gt_single_push(template, cid):
    """
    简单个推, 给单个用户推送

    :param: template: obj: 个推模板
    :param: cid: str: 个推唯一用户标识
    """
    obj = GtPush(template)
    res = obj.pushMessageToSingle(cid)
    return res


def gt_muti_push(template, cids, limit=50):
    """
    简单个推, 给多个用户推送
    
    :param: template: obj: 个推模板
    :param: cids: list: 个推唯一用户标识组成的列表
    :param: limit: int: 官方推荐列表推送值
    """
    if len(cids) <= limit:
        obj = GtPush(template)
        res = obj.pushMessageToList(cids)
        print(res)
    else:
        rg = len(cids) / limit
        num = rg if len(cids) % limit else rg + 1
        for i in range(num):
            single_cids = cids[i * limit:i * limit + limit]
            obj = GtPush(template)
            res = obj.pushMessageToList(single_cids)
            print(res)
    return True


def push_message_toapp(title, content, body, image='',
                       oslist=["ANDROID", "IOS"], plist=["上海"], taglist=[]):
    """
    按OS版本,区域,标签来进行推送
    """
    template = TransmissionTemplateDemo(title, content, body, image)
    obj = GtPush(template)
    obj.push_Message_ToApp(oslist, plist, taglist)
    print "OVER"
    return True


def gt_trans_push(cid_dics, title, content, body, image=''):
    """
    个推透传推送

    :param: cid_dics: list: 个推唯一用户标识组成的列表, 里面包含字典数据
        例如:
            cid_dics = [{"cid": "xxx", "os": "iOS"}, {"cid": "xxx", "os": "android"}]

    :param: title: str: 透传标题, 通知栏/透传标题一样
    :param: content: str: 透传内容, 可以为json字符串
        例如:
            content = "{
            "type": 6000,
            "title": "will",
            "content": "will",
            "extra": {
                "event_type": 1100,
                "event_url": "http://www.imxingzhe.com",
                "pic_url": "http://www.imxingzhe.com/1.png"
                }
            }"
    :param: body: str: 通知栏推送的消息内容
    :param: image: str: 通知栏消息图片, 默认没有
    """

    if cid_dics and len(cid_dics) == 1:
        # 如果没有给或者没有拿到os版本, 那么默认为other
        # 意味着不会推送APN消息, 这一点在TransmissionTemplateDemo模板中已过滤
        os = cid_dics[0].get("os", "other")
        cid = cid_dics[0].get("cid", "")
        print("os:{}".format(os))
        template = TransmissionTemplateDemo(title, content, body, image, os)
        res = gt_single_push(template, cid)
        print("透传结果:{}".format(res))
        # if os == "android":
        #    # android的通知栏推送要另外处理, 也就是做透传推送, 并且要做一个通知栏推送
        #    notification_template = NotificationTemplateDemo(title, body)
        #    res = gt_single_push(notification_template, cid)
        #    print("安卓通知栏结果:{}".format(res))
    else:
        # 由于iOS和android的推送方式不一样, 所以分iOS和android推送
        # 默认先推送IOS,再推送安卓,再推送其他
        ios_lis = []
        android_lis = []
        other_lis = []
        for dic in cid_dics:
            os = dic.get("os", "other")
            cid = dic.get("cid", "")
            if os and cid:
                if os == "android":
                    android_lis.append(cid)
                elif os == "iOS":
                    ios_lis.append(cid)
                else:
                    other_lis.append(cid)
            else:
                raise KwargsParameterError('cid')

        if ios_lis:
            template = TransmissionTemplateDemo(title, content, body, image, "iOS")
            res = gt_muti_push(template, ios_lis)
            print("iOS推送完毕")

        if android_lis:
            template = TransmissionTemplateDemo(title, content, body, image, "android")
            res = gt_muti_push(template, android_lis)
            # android的通知栏推送要另外处理
            # template = NotificationTemplateDemo(title, body)
            # res = gt_muti_push(template, android_lis)
            print("android推送完毕")

        if other_lis:
            template = TransmissionTemplateDemo(title, content, body, image, "other")
            res = gt_muti_push(template, other_lis)
            print("不知道版本机器推送完毕")

    return True
