设计模式专题-02-工厂模式/策略模式

Posted April 07, 2017

工厂模式

场景

场景01: 假设我们对动物这个类抽象,动物类为基类，然后猫和狗为子类，它们都有同样的方法move(); 我们肯定不希望程序运行过程中对动物类的所有子类都去实例化然后在调用move方法,而是用类当参数,实现实例化的对象直接调用move方法,该方法就是对应类的move方法;


场景02: 比如说我们对业务做一些监控,去数据库中执行sql然后将结果发送报表出来;一个监控还好,我们写一个执行SQL,一个格式话数据,外加一个发送报表和监控的函数就OK了; 但是,问题在于企业业务监控觉不是一个2个,那么此时我们就能使用工厂模式来减少工作量;我们可以定义一个工厂类,然后每个监控是一个类,调用工厂类的方法去实例化监控类,然后都统一调用report 方法即可;

实现逻辑

实现逻辑: 核心思想是定义一个工厂类(其实就是一个入口类)，将对应的函数或者是类当做参数传到工厂类中，然后返回特定对象的实例 或者 其他操作

File Name: 工厂模式.py
Author: WillDX
mail: xiang.dai@shuyun.com
Created Time: 三  7/ 6 18:00:39 2016
'''
class Ab:
    a=3
class Ac:
    a=0

class MyFactory:
    def get_instance(self,ins):
        return ins()

if __name__=="__main__":
    mf = MyFactory()  #mf是实例
    print(type(mf.get_instance(Ab)))
    print(type(mf.get_instance(Ac)))
    obj = mf.get_instance(Ab) #obj为新的shili
    print obj.a
    obj = mf.get_instance(Ac)
    print obj.a

import sys
import json
from react import re_act
from getapi import get_api
from muti import Consumer
from report import create_standard,create_report
from handle_config import HandleConfig
from godbox.utils.jsonencoder import JSONEncoder

reload(sys)
sys.setdefaultencoding('utf-8')

class Factory(object):
    def get_instance(self,cls):
        return cls()

class Base(object):
    """发送报表基类"""

    def __init__(self):
        self.srm_id = 3
        self.content = None
    
    def alert(self):
        """发送报警"""
        res = create_standard(**self.__dict__)
        print "通知返回:",res

    def report(self):
        """发送报表"""
        self.raws = self.func_source()
        if self.raws:
            url,html = create_report(**self.__dict__)
            self.content = html
            print "[报表地址]:",url
            self.alert()
        else:
            print "{}监控正常".format(self.class_name)

    def func_source(self):
        """将source放到func中执行"""
        produce = ( i for i in self.source if i)
        res = Consumer.do(produce,self.func,10)
        return res

    @property
    def source(self):
        """获取源数据:id,service_name,environment"""
        try:
            info = get_api(self.api_info["apps_api"],self.api_info)
        except Exception as e:
            print("ERROR:",e)
            self.content = "[微服务组件]-{}接口调不通".format(self.api_info["apps_api"])
            self.alert #接口调不通告警
            sys.exit()

        data = info["apps"]
        lis = []
        for i in data:
            dic = {
                "id":i["id"],
                "service_name":i["env"]["SERVICE_NAME"],
                "environment":i["env"]["ENVIRONMENT"]}
            lis.append(dic)

        # 存储service_name列表
        this_slis = [ i["service_name"] for i in lis if i]
        hc = HandleConfig("config.ini")
        try:
            # 判断apps差值
            old_slis = hc.get_section_option_value(self.class_name,"service_name")
            diff = list(set(eval(old_slis)).difference(set(this_slis)))
            if diff:
                print("diff:",diff)
                self.content = "[微服务组件]-{}不存在,请确认是否正常下线".format(diff)
                self.alert
                sys.exit()
        except Exception as e:
            print("ERROR:",e)
            hc.add_section(self.class_name)
        hc.write_section_option(self.class_name,"service_name",this_slis)
        return lis

        
    def func(self,**kw):
        """单次操作"""
        try:
            url = self.api_info["app_status_api"].format(kw["id"])
            res = get_api(url,self.api_info)
            status = res["app"]["tasksRunning"] 
            if int(status) <= 0:
                lis = [kw["environment"],kw["service_name"],kw["id"],"tasksRunning状态异常"]
                return [lis,]
        except Exception as e:
            print("ERROR:",e)
            self.content = "[微服务组件]-{}接口调不通".format(url)
            lis = [kw["id"],kw["service_name"],kw["environment"],e]
            return lis

# -*- coding: utf-8 -*-
'''
File Name: 策略模式_class.py
Author: WillDX
mail: xiang.dai@shuyun.com
Created Time: 三  7/ 6 22:41:50 2016
'''
class Moveable:
    def move(self):
        print('Move...')

class MoveOnFeet(Moveable):
    def move(self):
        print("Move on Feet.")
        
class MoveObj:
    def set_move(self,moveable):
        self.moveable = moveable()

    def move(self):
        self.moveable.move()

if __name__ == '__main__':
    m = MoveObj()
    m.set_move(Moveable)
    m.move()
    m.set_move(MoveOnFeet)
    m.move()
