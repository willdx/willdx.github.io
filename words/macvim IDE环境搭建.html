function RunPython()
  let mp = &makeprg
  let ef = &errorformat
  let exeFile = expand("%:t")
  setlocal makeprg=python\ -u
  set efm=%C\ %.%#,%A\ \ File\ \"%f\"\\,\ line\ %l%.%#,%Z%[%^\ ]%\\@=%m
  silent make %
  copen
  let &makeprg = mp
  let &errorformat = ef
endfunction


"新建.c,.h,.sh,.java文件，自动插入文件头
autocmd BufNewFile *.py,*.sh,*.md,*.cpp,*.[ch],*.java exec ":call SetTitle()"
""定义函数SetTitle，自动插入文件头
func SetTitle()
    "如果文件类型为.sh文件
    if &filetype == 'sh'
        call setline(1,"\#!/bin/bash")
        call append(line("."), "\# File Name: ".expand("%"))
        call append(line(".")+1, "\# Author: willdx")
        call append(line(".")+2, "\# mail: daixiang1992@gmail.com")
        call append(line(".")+3, "\# Created Time: ".strftime("%c"))
        call append(line(".")+4, "")
    elseif &filetype == 'python'
        call setline(1,"\# -*- coding: utf-8 -*-")
        call append(line("."), "'''")
        call append(line(".")+1, "File Name: ".expand("%"))
        call append(line(".")+2, "Author: willdx")
        call append(line(".")+3, "mail: daixiang1992@gmail.com")
        call append(line(".")+4, "Created Time: ".strftime("%c"))
        call append(line(".")+5, "'''")
        call append(line(".")+6, "")
    " c相关
    elseif &filetype == 'cpp'
        call append(line(".")+6, "#include<iostream>")
        call append(line(".")+7, "using namespace std;")
        call append(line(".")+8, "")
    elseif &filetype == 'c'
        call append(line(".")+6, "#include<stdio.h>")
        call append(line(".")+7, "")
    else
        call setline(1,"---")
        call setline(2, "title: ".expand("%"))
        call setline(3, "date: ".expand("%c"))
        call setline(4, "tags: ")
        call setline(5, "categories: ")
        call setline(6, "---")
        call setline(7, "")
    endif
endfunc

func! CompileRunGcc()
    exec "w"
    if &filetype == 'c'
        exec "!g++ % -o %<"
        exec "! ./%<"
    elseif &filetype == 'cpp'
        exec "!g++ % -o %<"
        exec "! ./%<"
    elseif &filetype == 'java' 
        exec "!javac %" 
        exec "!java %<"
    elseif &filetype == 'sh'
        :!./%
    endif
endfunc

func! Rungdb()

2.常用快捷键

内容 快捷键及其描述
1.快速移动 ;;f内容 向下跳转某个字符 ;;ge内容 向上跳转某个字符 ;j 向下跳转某行 ;k 向上跳转某行
2.返回快捷键 Ctrl+o 返回光标上次停留位置 Ctrl+i 返回光标 Ctrl+] 标签入栈 ;tn 下一个标签 ;tp 上一个标签
3. 常用面板 ;fl 显示当前目录文件 ;bl 显示buffer文件 ;tb 显示tagbar
4. 选择复制 双击开始的行,鼠标向下滑动自动选中,Ctrl+c复制
5. 快速注释与取消注释 ;cc 单行快速注释 ;cu 单行快速取消注释 数字键;cc 多行快速注释 数字键;cu 多行快速取消注释
6. pep8格式化代码并且执行 Python ;8 格式化代码,若有错误跳转到错误行
7.代码折叠 zc 折叠 zo 取消折叠
